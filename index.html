function goHome() {  
document.querySelectorAll('.view-container').forEach(v => v.classList.add('hidden'));  
document.getElementById('view-menu').classList.remove('hidden');  
updateProg(0, 1);  
}
function startLec(lecName) {  
pool = ALL_Q.map((q, i) => q.lec === lecName ? i : -1).filter(i => i !== -1);  
initQuiz();  
}
function startAll() {  
pool = ALL_Q.map((_, i) => i);  
initQuiz();  
}
function initQuiz() {  
qIdx = 0; correct = 0; wrong = 0; wrongIndices = [];  
document.querySelectorAll('.view-container').forEach(v => v.classList.add('hidden'));  
document.getElementById('view-quiz').classList.remove('hidden');  
loadQ();  
}
function loadQ() {  
const qData = ALL_Q[pool[qIdx]];  
document.getElementById('q-meta').textContent = `${qData.lec} | 
ال
لﺴﻮ
ا
 ${qIdx + 1} 
ﻣﻦ
 
${pool.length}`;  
document.getElementById('q-text').textContent = qData.q;  
document.getElementById('q-tr').textContent = qData.tr;  
document.getElementById('q-tr').classList.remove('show');  
document.getElementById('feedback').classList.remove('show');  
document.getElementById('btn-next').classList.add('hidden');  
const optsCont = document.getElementById('opts');  
optsCont.innerHTML = '';  
qData.opts.forEach((o, i) => {  
const div = document.createElement('div');  
div.className = 'opt';  
div.innerHTML = `<strong>${String.fromCharCode(65+i)}</strong> <span>${o}</span>`;  
div.onclick = () => checkAns(i);  
optsCont.appendChild(div);  
});  
updateProg(qIdx, pool.length);  
}
function checkAns(idx) {  
const qData = ALL_Q[pool[qIdx]];  
const opts = document.querySelectorAll('.opt');  
opts.forEach(o => o.classList.add('locked'));  
if(idx === qData.ans) {  
correct++;  
opts[idx].classList.add('correct');  
showFeedback("
أﺣﺴﻨﺖ
 !
ﺻﺤﻴﺤﺔ
 
إﺟاﺑﺔ
.", "var(--correct)");  
} else {  
wrong++;  
wrongIndices.push(pool[qIdx]);  
opts[idx].classList.add('wrong');  
opts[qData.ans].classList.add('correct');  
showFeedback(`
ﻫﻲ
 
لﺼﺤﻴﺤﺔ
ا
 
ﻹﺟاﺑﺔ
ا
 .
أ
ﺧﻄ
: ${qData.opts[qData.ans]}`, "var(--wrong)");  
}
document.getElementById('btn-next').classList.remove('hidden');  
}
function showFeedback(txt, color) {  
const fb = document.getElementById('feedback');  
fb.textContent = txt;  
fb.style.color = color;  
fb.classList.add('show');  
}
function nextQ() {  
qIdx++;  
if(qIdx < pool.length) loadQ();  
else showResults();  
}
function showResults() {  
document.querySelectorAll('.view-container').forEach(v => v.classList.add('hidden'));  
document.getElementById('view-results').classList.remove('hidden');  
document.getElementById('res-score').textContent = `${Math.round((correct/pool.length)*100)}%`;  
document.getElementById('res-stats').textContent = `
لﺼﺢ

: ${correct} | 
أ
لﺨﻄ

document.getElementById('wrong-section').classList.remove('hidden');  
} else {  
document.getElementById('wrong-section').classList.add('hidden');  
}
updateProg(pool.length, pool.length);  
}
function retryWrong() {  
pool = [...wrongIndices];  
initQuiz();  
}
function updateProg(current, total) {  
const pct = Math.round((current/total)*100);  
document.getElementById('prog-fill').style.width = pct + '%';  
document.getElementById('prog-pct').textContent = pct + '%';  
}
